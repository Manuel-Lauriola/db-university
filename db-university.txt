--SELECT

1. Selezionare tutti gli studenti nati nel 1990
SELECT `name`,`surname`,`date_of_birth`
FROM `students` WHERE YEAR (`date_of_birth`) = 1990; (160)

2. Selezionare tutti i corsi che valgono più di 10 crediti
SELECT `name`, `cfu`
FROM `courses` WHERE (`cfu`) > 10 (479)

3. Selezionare tutti gli studenti che hanno più di 30 anni
SELECT `name`, `surname`, `date_of_birth`
FROM `students` WHERE YEAR (`date_of_birth`) < 1995 (1000?)

4. Selezionare tutti i corsi del primo semestre del primo anno di un qualsiasi corso di
laurea
SELECT `name`, `period`
FROM `courses`WHERE (`period`) = "I semestre" AND (`year`) = 1 (286)

5. Selezionare tutti gli appelli d'esame che avvengono nel pomeriggio (dopo le 14) del
20/06/2020
SELECT `course_id`, `hour`
FROM `exams`WHERE hour (`hour`) >= 14 AND DATE(`date`) = "2020-06-20"

6. Selezionare tutti i corsi di laurea magistrale
SELECT `name`, `level`
FROM `degrees`WHERE (`level`) = "magistrale"

7. Da quanti dipartimenti è composta l'università?
SELECT COUNT(*)
FROM `departments` (12)

8. Quanti sono gli insegnanti che non hanno un numero di telefono?
SELECT `name`, `surname`
FROM `teachers`WHERE (`phone`) IS NULL
